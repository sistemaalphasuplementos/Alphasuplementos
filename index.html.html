Para o app buscar os dados da sua planilha do Google Sheets, siga os passos abaixo:

1
Abra sua planilha no Google Sheets e clique em Extensões → Apps Script
2
Cole o script fornecido no final desta página e clique em Implantar → Novo Implantação → App da Web
3
Em "Quem tem acesso" escolha "Qualquer pessoa" e copie a URL gerada
4
Cole a URL abaixo e clique em Salvar
Cole aqui a URL gerada no passo 3

Script para colar no Apps Script:

function doGet(e) { const action = e.parameter.action; const sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName('Página1'); if (action === 'getAll') { const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).filter(r => r[0] !== ''); const result = rows.map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput('{"error":"action inválida"}') .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName('Página1'); if (data.action === 'addCliente') { const c = data.cliente; sheet.appendRow([ c.nome, c.whatsapp, c.produto, c.dataCompra, c.valor, c.diasUso, '', '', '', '', '', '', '', '', c.codigo, c.indicadoPor || '', '', '' ]); return ContentService .createTextOutput('{"ok":true}') .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput('{"error":"action inválida"}') .setMimeType(ContentService.MimeType.JSON); }
